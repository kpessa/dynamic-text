```javascript
// IMPACT: 
// - No offline support
// - Network dependency for core functionality
// - Variable load times (200ms-2s depending on network)
```

**Recommended Fix**:
```javascript
// Move to proper ES module loading with caching
const BABEL_CACHE_KEY = 'babel-standalone-v7.28.2'

async function loadBabelCached() {
  // Check IndexedDB cache first
  const cached = await getCachedBabel(BABEL_CACHE_KEY)
  if (cached) return cached
  
  // Fallback to CDN with retry logic
  const response = await fetch('https://unpkg.com/@babel/standalone/babel.min.js')
  const babel = await response.text()
  
  // Cache for offline use
  await cacheBabel(BABEL_CACHE_KEY, babel)
  return babel
}
```

```typescript
// 2. Query result caching with TTL
const queryCache = new Map<string, { data: any, expires: number }>()

async function getCachedQuery(query: Query, ttl = 300000) {
  const key = queryToString(query)
  const cached = queryCache.get(key)
  
  if (cached && Date.now() < cached.expires) {
    return cached.data // Cache hit
  }
  
  const result = await getDocs(query)
  queryCache.set(key, {
    data: result.docs.map(doc => ({ id: doc.id, ...doc.data() })),
    expires: Date.now() + ttl
  })
  
  return queryCache.get(key)!.data
}
```

```typescript
    // LCP - Largest Contentful Paint
    new PerformanceObserver((list) => {
      const entries = list.getEntries()
```
