      // Execute complex TPN calculation
      const result = await calculateCompleteTPN({
        population: scenario.population,
        weight: 2.5,
        ingredients: ['dextrose', 'aminoacids', 'lipids', 'electrolytes'],
        additives: ['insulin', 'heparin'],
        complexity: scenario.complexity
      });
      
      const duration = performance.now() - start;
      
      expect(duration).toBeLessThan(scenario.maxTime);
      expect(result.accuracy).toBeGreaterThan(99.9); // Medical precision
      expect(result.safetyChecks.passed).toBe(true);
    }
  });
  
  it('should handle concurrent medical calculations', async () => {
    // Simulate multiple healthcare providers using system simultaneously
    const concurrentCalculations = Array.from({ length: 10 }, (_, i) =>
